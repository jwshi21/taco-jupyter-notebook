function mouse(hover, color) {
    var array_names = ["pos_r", "pos_r", "idx_r", "pos_c", "pos_c", "idx_c", "vals"]
    for (var i = 0; i < hover.length; i++) {
        var id = hover[i];
        if (id !== "_") {
            document.getElementById(array_names[i] + id).style.background = color;
        }
    }
}
